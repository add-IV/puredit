filtered_students = students.select("age", pl.col("DateOfBirth").name.toLowerCase(), first_name="name").to_pandas()

filtered_lectures = lectures.select("credits", pl.col("Room").name.toLowerCase(), professor="lecturer").to_pandas()

---

From students export columns
    age.
        .
        DateOfBirth.
        take its name.
        converted to lower case.
    .
    name as first_name.
to Pandas into filtered_students.

From lectures export columns
    credits.
        .
        Room.
        take its name.
        converted to lower case.
    .
    lecturer as professor.
to Pandas into filtered_lectures.
